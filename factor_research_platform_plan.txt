### 3.1 因子注册中心 (factor_registry.py)

```python
"""
因子注册中心：统一管理所有因子，支持动态注册和组合
"""
from typing import Dict, List, Callable, Optional
from dataclasses import dataclass, field
import yaml
from pathlib import Path

@dataclass
class FactorInfo:
    """因子信息"""
    name: str
    description: str
    formula: str
    source: str  # 'paper', 'llm', 'manual'
    parameters: Dict = field(default_factory=dict)
    code_path: Optional[str] = None
    performance_metrics: Dict = field(default_factory=dict)

class FactorRegistry:
    """因子注册中心"""
    
    def __init__(self, factors_dir: str = "factors/custom_factors"):
        self.factors_dir = Path(factors_dir)
        self.factors: Dict[str, FactorInfo] = {}
        self.factor_groups: Dict[str, List[str]] = {}
        self._load_existing_factors()
    
    def register_factor(self, factor_info: FactorInfo):
        """注册新因子"""
        self.factors[factor_info.name] = factor_info
        self._save_factor_info(factor_info)
    
    def create_factor_group(self, group_name: str, factor_names: List[str]):
        """创建因子组合"""
        self.factor_groups[group_name] = factor_names
        self._save_factor_group(group_name, factor_names)
    
    def get_factor_group(self, group_name: str) -> List[FactorInfo]:
        """获取因子组合"""
        if group_name not in self.factor_groups:
            raise ValueError(f"因子组合 {group_name} 不存在")
        return [self.factors[name] for name in self.factor_groups[group_name]]
    
    def list_factors(self, source: Optional[str] = None) -> List[FactorInfo]:
        """列出所有因子，可按来源过滤"""
        if source:
            return [f for f in self.factors.values() if f.source == source]
        return list(self.factors.values())
    
    def _load_existing_factors(self):
        """加载已有因子"""
        if self.factors_dir.exists():
            for factor_file in self.factors_dir.glob("*.yaml"):
                with open(factor_file, 'r', encoding='utf-8') as f:
                    data = yaml.safe_load(f)
                    self.factors[data['name']] = FactorInfo(**data)
    
    def _save_factor_info(self, factor_info: FactorInfo):
        """保存因子信息"""
        self.factors_dir.mkdir(parents=True, exist_ok=True)
        with open(self.factors_dir / f"{factor_info.name}.yaml", 'w', encoding='utf-8') as f:
            yaml.dump(factor_info.__dict__, f, allow_unicode=True)
    
    def _save_factor_group(self, group_name: str, factor_names: List[str]):
        """保存因子组合"""
        groups_dir = Path("factors/factor_groups")
        groups_dir.mkdir(parents=True, exist_ok=True)
        with open(groups_dir / f"{group_name}.yaml", 'w', encoding='utf-8') as f:
            yaml.dump({'group_name': group_name, 'factors': factor_names}, f, allow_unicode=True)
```

### 3.2 因子组合器 (factor_combiner.py)

```python
"""
因子组合器：将多个因子组合成特征集，用于模型训练
"""
import pandas as pd
import numpy as np
from typing import List, Dict, Optional
from qlib.data import D
from qlib.contrib.data.handler import Alpha158
from factor_registry import FactorRegistry, FactorInfo

class FactorCombiner:
    """因子组合器"""
    
    def __init__(self, registry: FactorRegistry):
        self.registry = registry
    
    def build_custom_handler(self, factor_group_name: str, instruments: str = "csi300",
                            start_time: str = "2020-01-01", end_time: str = "2024-12-31"):
        """
        根据因子组合构建自定义 Qlib Handler
        
        Args:
            factor_group_name: 因子组合名称
            instruments: 股票池
            start_time: 开始时间
            end_time: 结束时间
        
        Returns:
            自定义 Handler 配置（可直接用于 Qlib 训练）
        """
        factors = self.registry.get_factor_group(factor_group_name)
        
        # 构建因子表达式列表
        factor_fields = {}
        for factor in factors:
            factor_fields[factor.name] = factor.formula
        
        # 构建 Qlib Handler 配置
        handler_config = {
            "class": "Alpha158",
            "module_path": "qlib.contrib.data.handler",
            "kwargs": {
                "start_time": start_time,
                "end_time": end_time,
                "fit_start_time": start_time,
                "fit_end_time": end_time,
                "instruments": instruments,
                # 添加自定义因子
                "config": [
                    {
                        "feature_name": name,
                        "feature_config": {
                            "expr": formula,
                            "name": name
                        }
                    }
                    for name, formula in factor_fields.items()
                ]
            }
        }
        
        return handler_config
    
    def merge_with_baseline(self, factor_group_name: str, baseline: str = "Alpha158"):
        """
        将自定义因子与基线因子合并
        
        Args:
            factor_group_name: 自定义因子组合
            baseline: 基线因子集（如 Alpha158, Alpha360）
        
        Returns:
            合并后的因子配置
        """
        factors = self.registry.get_factor_group(factor_group_name)
        
        merge_config = {
            "baseline": baseline,
            "custom_factors": [f.name for f in factors],
            "total_factor_count": 158 + len(factors)  # Alpha158 + 自定义
        }
        
        return merge_config
    
    def evaluate_factor_correlation(self, factor_names: List[str], 
                                   start_time: str = "2023-01-01",
                                   end_time: str = "2023-12-31"):
        """
        评估因子间相关性，避免高度相关因子重复
        
        Args:
            factor_names: 因子名称列表
            start_time: 开始时间
            end_time: 结束时间
        
        Returns:
            相关性矩阵
        """
        # 获取因子数据
        factor_data = {}
        for name in factor_names:
            factor_info = self.registry.factors[name]
            # 这里需要根据实际因子公式获取数据
            # 简化示例：
            df = D.features(["SH000300"], [factor_info.formula], 
                           start_time=start_time, end_time=end_time)
            factor_data[name] = df
        
        # 计算相关性矩阵
        combined_df = pd.concat(factor_data.values(), axis=1)
        correlation_matrix = combined_df.corr()
        
        return correlation_matrix
```

### 3.3 实验管理器 (experiment_manager.py)

```python
"""
实验管理器：管理因子组合训练实验，记录实验配置和结果
"""
import json
import yaml
from pathlib import Path
from datetime import datetime
from typing import Dict, List, Optional
from dataclasses import dataclass, field, asdict
import subprocess

@dataclass
class ExperimentConfig:
    """实验配置"""
    name: str
    description: str
    factor_group: str
    model_type: str  # 'xgboost', 'lightgbm', 'catboost', 'linear'
    dataset: str
    instruments: str
    train_start: str
    train_end: str
    valid_start: str
    valid_end: str
    test_start: str
    test_end: str
    rolling_type: str = "expanding"
    rolling_step: int = 60
    hyperparameters: Dict = field(default_factory=dict)

@dataclass
class ExperimentResult:
    """实验结果"""
    exp_name: str
    ic: float
    icir: float
    rank_ic: float
    rank_icir: float
    annualized_return: float
    max_drawdown: float
    sharpe_ratio: float
    information_ratio: float
    model_path: str
    timestamp: str

class ExperimentManager:
    """实验管理器"""
    
    def __init__(self, experiments_dir: str = "experiments"):
        self.experiments_dir = Path(experiments_dir)
        self.experiments_dir.mkdir(parents=True, exist_ok=True)
        self.experiments: Dict[str, Dict] = self._load_experiments()
    
    def create_experiment(self, config: ExperimentConfig) -> str:
        """
        创建新实验
        
        Args:
            config: 实验配置
        
        Returns:
            实验名称
        """
        exp_name = f"exp_{len(self.experiments) + 1:03d}_{config.name}"
        exp_dir = self.experiments_dir / exp_name
        exp_dir.mkdir(parents=True, exist_ok=True)
        
        # 保存配置
        with open(exp_dir / "config.yaml", 'w', encoding='utf-8') as f:
            yaml.dump(asdict(config), f, allow_unicode=True)
        
        # 记录实验
        self.experiments[exp_name] = {
            "config": asdict(config),
            "status": "created",
            "created_at": datetime.now().isoformat()
        }
        
        return exp_name
    
    def run_experiment(self, exp_name: str) -> ExperimentResult:
        """
        运行实验
        
        Args:
            exp_name: 实验名称
        
        Returns:
            实验结果
        """
        exp_dir = self.experiments_dir / exp_name
        config_file = exp_dir / "config.yaml"
        
        # 调用训练脚本
        cmd = [
            "python", "roll/roll.py", "model", "train",
            "--config", str(config_file),
            "--exp-name", exp_name
        ]
        
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        if result.returncode != 0:
            raise RuntimeError(f"实验运行失败: {result.stderr}")
        
        # 解析结果
        exp_result = self._parse_experiment_result(exp_name)
        
        # 更新实验状态
        self.experiments[exp_name]["status"] = "completed"
        self.experiments[exp_name]["result"] = asdict(exp_result)
        self._save_experiments()
        
        return exp_result
    
    def compare_experiments(self, exp_names: List[str]) -> pd.DataFrame:
        """
        对比多个实验结果
        
        Args:
            exp_names: 实验名称列表
        
        Returns:
            对比结果 DataFrame
        """
        results = []
        for name in exp_names:
            if name in self.experiments and "result" in self.experiments[name]:
                result_data = self.experiments[name]["result"]
                result_data["exp_name"] = name
                results.append(result_data)
        
        df = pd.DataFrame(results)
        return df
    
    def _load_experiments(self) -> Dict:
        """加载实验记录"""
        exp_file = self.experiments_dir / "experiments.json"
        if exp_file.exists():
            with open(exp_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {}
    
    def _save_experiments(self):
        """保存实验记录"""
        exp_file = self.experiments_dir / "experiments.json"
        with open(exp_file, 'w', encoding='utf-8') as f:
            json.dump(self.experiments, f, ensure_ascii=False, indent=2)
    
    def _parse_experiment_result(self, exp_name: str) -> ExperimentResult:
        """解析实验结果（从 Qlib workflow 中获取）"""
        # 这里需要从 Qlib 的 recorder 中读取实际结果
        # 简化示例：
        return ExperimentResult(
            exp_name=exp_name,
            ic=0.05,
            icir=0.3,
            rank_ic=0.04,
            rank_icir=0.25,
            annualized_return=0.15,
            max_drawdown=0.2,
            sharpe_ratio=0.75,
            information_ratio=0.5,
            model_path=str(self.experiments_dir / exp_name / "model.pkl"),
            timestamp=datetime.now().isoformat()
        )
```

### 3.4 模型对比器 (model_comparator.py)

```python
"""
模型对比器：对比不同模型/因子组合的性能
"""
import pandas as pd
import numpy as np
from typing import List, Dict, Optional
from pathlib import Path
import matplotlib.pyplot as plt
import seaborn as sns
from experiment_manager import ExperimentManager

class ModelComparator:
    """模型对比器"""
    
    def __init__(self, exp_manager: ExperimentManager):
        self.exp_manager = exp_manager
    
    def compare_models(self, exp_names: List[str], 
                      metrics: List[str] = None) -> pd.DataFrame:
        """
        对比多个模型
        
        Args:
            exp_names: 实验名称列表
            metrics: 要对比的指标列表
        
        Returns:
            对比结果 DataFrame
        """
        if metrics is None:
            metrics = ['ic', 'icir', 'rank_ic', 'rank_icir', 
                      'annualized_return', 'max_drawdown', 'sharpe_ratio']
        
        df = self.exp_manager.compare_experiments(exp_names)
        
        if df.empty:
            raise ValueError("没有找到实验结果")
        
        # 提取关键指标
        comparison = df[['exp_name'] + metrics].copy()
        
        # 计算排名
        for metric in metrics:
            if metric == 'max_drawdown':
                # 最大回撤越小越好
                comparison[f'{metric}_rank'] = comparison[metric].rank(ascending=True)
            else:
                # 其他指标越大越好
                comparison[f'{metric}_rank'] = comparison[metric].rank(ascending=False)
        
        # 综合排名
        rank_columns = [f'{m}_rank' for m in metrics]
        comparison['overall_rank'] = comparison[rank_columns].mean(axis=1).rank()
        
        return comparison.sort_values('overall_rank')
    
    def plot_comparison(self, comparison_df: pd.DataFrame, 
                       save_path: str = "experiments/comparison.png"):
        """
        绘制对比图表
        
        Args:
            comparison_df: 对比结果 DataFrame
            save_path: 保存路径
        """
        metrics = ['ic', 'icir', 'annualized_return', 'sharpe_ratio']
        
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        axes = axes.flatten()
        
        for i, metric in enumerate(metrics):
            ax = axes[i]
            comparison_df.plot.bar(x='exp_name', y=metric, ax=ax, legend=False)
            ax.set_title(f'{metric.upper()} Comparison')
            ax.set_xlabel('')
            ax.tick_params(axis='x', rotation=45)
        
        plt.tight_layout()
        plt.savefig(save_path, dpi=300, bbox_inches='tight')
        plt.close()
    
    def generate_comparison_report(self, exp_names: List[str],
                                  save_path: str = "experiments/report.md"):
        """
        生成对比报告
        
        Args:
            exp_names: 实验名称列表
            save_path: 报告保存路径
        """
        comparison = self.compare_models(exp_names)
        
        report = f"""# 模型对比报告

生成时间: {pd.Timestamp.now()}

## 对比实验

{', '.join(exp_names)}

## 性能对比

{comparison.to_markdown()}

## 最佳模型

根据综合排名，最佳模型是: **{comparison.iloc[0]['exp_name']}**

## 详细分析

### IC 分析
- 最高 IC: {comparison['ic'].max():.4f}
- 最低 IC: {comparison['ic'].min():.4f}
- 平均 IC: {comparison['ic'].mean():.4f}

### 风险收益分析
- 最高年化收益: {comparison['annualized_return'].max():.2%}
- 最大回撤最小: {comparison['max_drawdown'].min():.2%}
- 最高夏普比率: {comparison['sharpe_ratio'].max():.2f}

## 建议

根据以上分析，建议:
1. 使用 {comparison.iloc[0]['exp_name']} 作为主模型
2. 关注 {comparison.iloc[1]['exp_name']} 作为备选
3. 定期重新评估模型表现
"""
        
        with open(save_path, 'w', encoding='utf-8') as f:
            f.write(report)
        
        return report
```

## 五、使用流程示例

### 5.1 注册 LLM 研究出的因子

```bash
# 方式 1: 使用命令行工具
python scripts/register_factor.py \
    --name "momentum_20d" \
    --description "20日动量因子" \
    --formula "$close / Ref($close, 20) - 1" \
    --source "llm" \
    --parameters '{"window": 20}'

# 方式 2: 直接创建 YAML 文件
# factors/custom_factors/momentum_20d.yaml
```

### 5.2 创建因子组合

```python
from research.factor_registry import FactorRegistry

registry = FactorRegistry()

# 创建因子组合
registry.create_factor_group("llm_research_001", [
    "momentum_20d",
    "reversal_5d",
    "volume_price_trend",
    "volatility_adjusted_return"
])

# 查看因子列表
print(registry.list_factors(source="llm"))
```

### 5.3 运行对比实验

```bash
# 实验 1: 基线模型
python scripts/run_experiment.py \
    --factor-group baseline \
    --model xgboost \
    --exp-name exp_001_baseline

# 实验 2: LLM 因子组合
python scripts/run_experiment.py \
    --factor-group llm_research_001 \
    --model xgboost \
    --exp-name exp_002_llm_factors

# 实验 3: 不同模型对比
python scripts/run_experiment.py \
    --factor-group llm_research_001 \
    --model lightgbm \
    --exp-name exp_003_llm_lgbm
```

### 5.4 生成对比报告

```bash
python scripts/compare_models.py \
    --experiments exp_001_baseline,exp_002_llm_factors,exp_003_llm_lgbm \
    --output experiments/comparison_report.md \
    --plot experiments/comparison.png
```

### 5.5 查看结果

```python
from research.model_comparator import ModelComparator
from research.experiment_manager import ExperimentManager

exp_manager = ExperimentManager()
comparator = ModelComparator(exp_manager)

# 对比模型
comparison = comparator.compare_models([
    "exp_001_baseline",
    "exp_002_llm_factors",
    "exp_003_llm_lgbm"
])

print(comparison)

# 生成报告
comparator.generate_comparison_report(
    ["exp_001_baseline", "exp_002_llm_factors", "exp_003_llm_lgbm"],
    save_path="experiments/report.md"
)
```

## 七、常见问题与解决方案

### Q1: 如何验证因子表达式是否正确？

A: 使用 Qlib 的 D.features() 方法测试：

```python
from qlib.data import D

# 测试因子表达式
df = D.features(["SH000300"], ["$close / Ref($close, 10) - 1"], 
               start_time="2023-01-01", end_time="2023-01-31")
print(df.head())
```

### Q2: 如何处理因子缺失值？

A: 在因子注册时添加预处理配置：

```yaml
preprocessing:
  drop_na: true
  fillna: 0
  winsorize:
    method: percentile
    lower: 0.025
    upper: 0.975
  normalize:
    method: zscore
    cross_section: true
```

### Q3: 如何加速因子计算？

A: 
1. 使用 Qlib 的缓存机制
2. 避免在因子表达式中使用循环
3. 使用向量化操作
4. 考虑使用 Dask 进行并行计算

### Q4: 如何对比不同时间段的模型表现？

A: 使用 ExperimentManager 的时间序列分析功能：

```python
exp_manager.analyze_time_series(
    exp_names=["exp_001", "exp_002"],
    frequency="monthly"
)
```

### Q5: 如何将模型部署到生产环境？

A: 
1. 使用 model_comparator 选择最佳模型
2. 导出模型到 model_pkl/ 目录
3. 更新 roll/config.yaml 中的模型配置
4. 通过 GitHub Actions 自动部署
